% Terminology config
\ExplSyntaxOn

\msg_new:nnn { report } { missing-term }
  { warning:~the~term~'#1'~is~not~defined! }

\cs_new:Nn \report_apply_case:nn {
  \str_case_e:nnF { #1 }
    {
      { \text_uppercase:n { #1 } } { \text_uppercase:n { #2 } }
      { \text_lowercase:n { #1 } } { #2 }
      { \text_titlecase_first:n { \text_lowercase:n {#1} } }
        { \text_titlecase_first:n { #2 } }
      { \text_titlecase_all:n { \text_lowercase:n {#1} } }
        { \text_titlecase_all:n { #2 } }
    }
    { #1 }
}

\cs_new:Nn \report_handle_tr:nn {
  \cs_if_exist:cTF { \text_lowercase:n {#1}-#2 }
    { \report_apply_case:nn { #1 } { \use:c { \text_lowercase:n {#1}-#2 } } }
    {
      \msg_warning:nne { report } { missing-term }
      { \text_lowercase:n {#1}-#2 }
      #1
    }
}

\NewExpandableDocumentCommand{\tr}{ O{vn} m }
  {
    \texorpdfstring
     { \report_handle_tr:nn {#2} {#1} }
     { \text_purify:n { \report_handle_tr:nn {#2} {#1} } }
  }

\NewDocumentCommand{\trdef}{ O{vn} m m }
{
  \exp_args:Nc \tl_set:Nn {#2-#1} {#3}
}
